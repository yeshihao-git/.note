# 抽象工厂模式

**what：**
一种创建型设计模式，它能创建一系列相关的对象，而无需指定其具体类
1. 一个工厂生成多种产品

![[Pasted image 20260617161153.png]]

**how**：
1. 抽象工厂依赖抽象产品
2. 多个抽象产品，分别派生多个具体产品
3. 一个抽象工厂派生多个具体工厂
4. 每个具体工厂生产多个具体产品

```cpp
#include <string>
#include <iostream>
#include <memory>

// AbstractProductA
class ProductA
{
public:
    virtual ~ProductA() = default;
    virtual std::string someOperation() = 0;
};

// ConcreteProductA1
class ConcreteProductA1 : public ProductA
{
public:
    std::string someOperation() override
    {
        return "ConcreteProductA1";
    }
};

// ConcreteProductA2
class ConcreteProductA2 : public ProductA
{
public:
    std::string someOperation() override
    {
        return "ConcreteProductA2";
    }
};

// AbstractProductB
class ProductB
{
public:
    virtual ~ProductB() = default;
    virtual std::string someOperation() = 0;
};

// ConcreteProductB1
class ConcreteProductB1 : public ProductB
{
public:
    std::string someOperation() override
    {
        return "ConcreteProductB1";
    }
};

// ConcreteProductB2
class ConcreteProductB2 : public ProductB
{
public:
    std::string someOperation() override
    {
        return "ConcreteProductB2";
    }
};

// AbstractFactory
class Factory
{
public:
    virtual ~Factory() = default;
    virtual std::unique_ptr<ProductA> createProductA() = 0;
    virtual std::unique_ptr<ProductB> createProductB() = 0;
};

// ConcreteFactory1
class ConcreteFactory1 : public Factory
{
public:
    std::unique_ptr<ProductA> createProductA() override
    {
        return std::make_unique<ConcreteProductA1>();
    }

    std::unique_ptr<ProductB> createProductB() override
    {
        return std::make_unique<ConcreteProductB1>();
    }
};

// ConcreteFactory2
class ConcreteFactory2 : public Factory
{
public:
    std::unique_ptr<ProductA> createProductA() override
    {
        return std::make_unique<ConcreteProductA2>();
    }

    std::unique_ptr<ProductB> createProductB() override
    {
        return std::make_unique<ConcreteProductB2>();
    }
};

// Client
void clientCode(std::unique_ptr<Factory> factory)
{
    std::unique_ptr<ProductA> productA = factory->createProductA();
    std::cout << productA->someOperation() << std::endl;
    std::unique_ptr<ProductB> productB = factory->createProductB();
    std::cout << productB->someOperation() << std::endl;
}

int main()
{
    std::unique_ptr<ConcreteFactory1> factory1 = std::make_unique<ConcreteFactory1>();
    clientCode(std::move(factory1));
    std::unique_ptr<ConcreteFactory2> factory2 = std::make_unique<ConcreteFactory2>();
    clientCode(std::move(factory2));
}
```
